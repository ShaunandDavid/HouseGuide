--- src/utils/classify.js (create app/src/utils/classify.js) ---
const WU = ["write up","write-up","incident","violation","warning","policy","consequence"];
const CM = ["commitment","goal","plan","amends","step","sponsor","mentor","aa","na","meeting"];
export function classifyByKeywords(txt){
  const t = (txt||'').toLowerCase();
  const score = {writeup:0, commitment:0};
  WU.forEach(k=>{ if(t.includes(k)) score.writeup++; });
  CM.forEach(k=>{ if(t.includes(k)) score.commitment++; });
  if(score.writeup===0 && score.commitment===0) return {label:null, confidence:0};
  const label = score.writeup>score.commitment? 'writeup':'commitment';
  const confidence = Math.max(score.writeup, score.commitment)/(score.writeup+score.commitment);
  return {label, confidence};
}

--- src/utils/report.js (create app/src/utils/report.js) ---
export function weeklyTemplate(resident, weekStart, entries){
  const pick = (k)=> (entries?.[k]?.trim?.() || 'N/A');
  const d = new Date(weekStart);
  const weekStr = isNaN(d) ? weekStart : d.toLocaleDateString();
  return `Resident: ${resident.firstName} ${resident.lastInitial}.  Week of: ${weekStr}

__Sponsor/Mentor:__ ${pick('s1_sponsor')}

__Work/School:__ ${pick('s2_work')}

__Chores/Compliance:__ ${pick('s3_chores')}

__Demeanor / Participation:__ ${pick('s4_demeanor')}

__Professional Help / Appointments:__ ${pick('s5_professional')}
`;
}

--- src/pages/Resident.jsx (create app/src/pages/Resident.jsx) ---
import React,{useEffect,useState} from 'react'
import { useParams } from 'react-router-dom'
import { pb } from '../api'
import Tesseract from 'tesseract.js'
import { classifyByKeywords } from '../utils/classify'
import { weeklyTemplate } from '../utils/report'

export default function Resident(){
  const {id}=useParams()
  const [resident,setResident]=useState(null)
  const [files,setFiles]=useState([])
  const [week,setWeek]=useState(() => new Date(new Date().setDate(new Date().getDate()-new Date().getDay()+1)).toISOString().slice(0,10)) // Monday
  const [sections,setSections]=useState({})
  const [busy,setBusy]=useState(false)

  useEffect(()=>{ (async()=>{
    const r = await pb.collection('residents').getOne(id)
    setResident(r)
    const fl = await pb.collection('files').getFullList({ filter: `resident = "${id}"` , sort:'-created'})
    setFiles(fl)
  })() },[id])

  async function handleImage(e){
    const file = e.target.files?.[0]; if(!file) return;
    setBusy(true)
    const imgUrl = URL.createObjectURL(file)
    const { data:{ text } } = await Tesseract.recognize(imgUrl, 'eng')
    const {label,confidence}= classifyByKeywords(text)
    let type = label
    if(!type || confidence < 0.6){
      type = window.confirm('Looks like COMMITMENT. OK = Commitment, Cancel = Write-Up') ? 'commitment' : 'writeup'
    }
    const fd = new FormData()
    fd.append('resident', id)
    fd.append('type', type)
    fd.append('image', file)
    fd.append('ocrText', text)
    await pb.collection('files').create(fd)
    const fl = await pb.collection('files').getFullList({ filter: `resident = "${id}"` , sort:'-created'})
    setFiles(fl); setBusy(false)
  }

  async function copyWeekly(){
    const entry = await pb.collection('reports').getFirstListItem(`resident = "${id}" && weekStart = "${week}"`,{requestKey: undefined}).catch(()=>null)
    const txt = weeklyTemplate(resident, week, entry||sections)
    await navigator.clipboard.writeText(txt)
    alert('Weekly report copied to clipboard')
  }

  return (
    <div style={{padding:16,fontFamily:'system-ui'}}>
      <h2>{resident? `${resident.firstName} ${resident.lastInitial}.` : '...'} — Weekly</h2>

      <div style={{margin:'12px 0',display:'grid',gap:8}}>
        <label>Week start (Mon): <input type="date" value={week} onChange={e=>setWeek(e.target.value)} /></label>
        <input placeholder="Sponsor/Mentor" onChange={e=>setSections(s=>({...s,s1_sponsor:e.target.value}))} />
        <input placeholder="Work/School" onChange={e=>setSections(s=>({...s,s2_work:e.target.value}))} />
        <input placeholder="Chores/Compliance" onChange={e=>setSections(s=>({...s,s3_chores:e.target.value}))} />
        <input placeholder="Demeanor/Participation" onChange={e=>setSections(s=>({...s,s4_demeanor:e.target.value}))} />
        <input placeholder="Professional Help/Appointments" onChange={e=>setSections(s=>({...s,s5_professional:e.target.value}))} />
        <div style={{display:'flex',gap:8}}>
          <button onClick={copyWeekly}>Copy Weekly</button>
          <label style={{border:'1px solid #ccc',padding:'8px',cursor:'pointer'}}>
            {busy? 'Scanning…' : 'Scan/Upload Photo'}
            <input type="file" accept="image/*" capture="environment" onChange={handleImage} style={{display:'none'}} />
          </label>
        </div>
      </div>

      <h3>Files</h3>
      <ul>
        {files.map(f=> (
          <li key={f.id}>
            [{f.type}] <a href={`${pb.baseUrl}/api/files/files/${f.id}/${f.image}`} target="_blank" rel="noreferrer">view</a>
          </li>
        ))}
      </ul>
    </div>
  )
}
